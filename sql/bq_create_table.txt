# 2. 建立新表（欄位已轉 STRING + 加上時間預設值）
CREATE TABLE `REVIEW.reviews`
(
    review_id STRING NOT NULL
        OPTIONS(description="評論編號(from apify)"),

    restaurant_id STRING NOT NULL
        OPTIONS(description="餐廳編號"),

    review_score INT64 NOT NULL
        OPTIONS(description="總評分1~5"),

    review_content STRING
        OPTIONS(description="評論內容"),

    # 💡 這三個欄位改成 STRING 以包容空值
    food_score STRING
        OPTIONS(description="評分-食物"),

    service_score STRING
        OPTIONS(description="評分-服務"),

    atmosphere_score STRING
        OPTIONS(description="評分-環境氣氛"),

    # 💡 加上 DEFAULT 讓 BigQuery 自動生成當下時間
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
        OPTIONS(description="資料建立時間"),

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
        OPTIONS(description="資料更新時間")
)
PARTITION BY DATE(created_at)
CLUSTER BY restaurant_id, review_score;
